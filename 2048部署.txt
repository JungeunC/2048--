python3 --version

# 安装python3-pip
apt install python3-pip

# 安装python3-venv
apt install python3-venv
# 创建虚拟环境
python3 -m venv venv
# 激活虚拟环境
source venv/bin/activate

# 在虚拟环境中安装依赖
pip install -r requirements.txt

mkdir -p logs

# 在虚拟环境中运行
# 使用nohup后台运行
nohup python3 2048.py > logs/app.log 2>&1 &

# 查看进程
ps aux | grep 2048.py

# 测试访问
curl http://localhost:5000
